046    /**
047     * Holds objects to be indexed.
048     * <br/>
049     * It looks like we need to keep object references in memory until they indexed properly.
050     * If indexing fails, all failed objects are retried. Still no support for max number of retries (todo)
051     * NOTE: if cluster state is RED, everything will probably fail and keep retrying forever.
052     * NOTE: This is shared class, so need to be thread-safe.
053     */
054    public class IndexRequestQueue implements InitializingBean {
055    
056        private static final Logger LOG = Logger.getLogger(IndexRequestQueue.class);
057    
058        private JSONDomainFactory jsonDomainFactory;
059        private ElasticSearchContextHolder elasticSearchContextHolder;
060        private Client elasticSearchClient;
061        private HibernatePersistenceContextInterceptor persistenceInterceptor;
062        private SessionFactory sessionFactory;
063    
064        /**
065         * A map containing the pending index requests.
066         */
067        Map<IndexEntityKey, Object> indexRequests = new HashMap<IndexEntityKey, Object>();
068    
069        /**
070         * A set containing the pending delete requests.
071         */
072        Set<IndexEntityKey> deleteRequests = new HashSet<IndexEntityKey>();
073    
074        /**
075         * No-args constructor.
076         */
077        public IndexRequestQueue() {
078        }
079    
080        public void setJsonDomainFactory(JSONDomainFactory jsonDomainFactory) {
081            this.jsonDomainFactory = jsonDomainFactory;
082        }
083    
084        public void setElasticSearchContextHolder(ElasticSearchContextHolder elasticSearchContextHolder) {
085            this.elasticSearchContextHolder = elasticSearchContextHolder;
086        }
087    
088        public void setElasticSearchClient(Client elasticSearchClient) {
089            this.elasticSearchClient = elasticSearchClient;
090        }
091    
092        public void setSessionFactory(SessionFactory sessionFactory) {
093            this.sessionFactory = sessionFactory;
094        }
095    
096        /**
097         */
098        public void afterPropertiesSet() throws Exception {
099            Assert.notNull(sessionFactory);
100            persistenceInterceptor = new HibernatePersistenceContextInterceptor();
101            persistenceInterceptor.setSessionFactory(sessionFactory);
102            persistenceInterceptor.setReadOnly();
103        }
104    
105        public void addIndexRequest(Object instance) {
106            addIndexRequest(instance, null);
107        }
108    
109        public void addIndexRequest(Object instance, Serializable id) {
110            synchronized (this) {
111                IndexEntityKey key = id == null ? new IndexEntityKey(instance) :
112                        new IndexEntityKey(id.toString(), GrailsHibernateUtil.unwrapIfProxy(instance).getClass());
113                indexRequests.put(key, GrailsHibernateUtil.unwrapIfProxy(instance));
114            }
115        }
116    
117        public void addDeleteRequest(Object instance) {
118            synchronized (this) {
119                deleteRequests.add(new IndexEntityKey(instance));
120            }
121        }
122    
123        public XContentBuilder toJSON(Object instance) {
124            try {
125                return jsonDomainFactory.buildJSON(instance);
126            } catch (Exception e) {
127                throw new IndexException("Failed to marshall domain instance [" + instance + "]", e);
128            }
129        }
130    
131        /**
132         * Execute pending requests and clear both index & delete pending queues.
133         */
134        public void executeRequests() {
135            Map<IndexEntityKey, Object> toIndex = new LinkedHashMap<IndexEntityKey, Object>();
136            Set<IndexEntityKey> toDelete = new HashSet<IndexEntityKey>();
137    
138            // Copy existing queue to ensure we are interfering with incoming requests.
139            synchronized (this) {
140                toIndex.putAll(indexRequests);
141                toDelete.addAll(deleteRequests);
142                indexRequests.clear();
143                deleteRequests.clear();
144            }
145    
146            // If there are domain instances that are both in the index requests & delete requests list,
147            // they are directly deleted.
148            toIndex.keySet().removeAll(toDelete);
149    
150            if (toIndex.isEmpty() && toDelete.isEmpty()) {
151                return;
152            }
153    
154            BulkRequestBuilder bulkRequestBuilder = new BulkRequestBuilder(elasticSearchClient);
155    
156            // Execute index requests
157            for (Map.Entry<IndexEntityKey, Object> entry : toIndex.entrySet()) {
158                SearchableClassMapping scm = elasticSearchContextHolder.getMappingContextByType(entry.getKey().getClazz());
159    //            XContentBuilder json = toJSON(entry.getValue());
160    //            byte[] json = (byte[]) entry.getValue();
161                persistenceInterceptor.init();
162                try {
163                    Session session = SessionFactoryUtils.getSession(sessionFactory, true);
164                    Object entity = entry.getValue();
165                    session.lock(entity, LockMode.NONE);
166                    XContentBuilder json = toJSON(entity);
167    
168                    bulkRequestBuilder.add(
169                            new IndexRequest(scm.getIndexName())
170                                    .type(scm.getElasticTypeName())
171                                    .id(entry.getKey().getId())      // how about composite keys?
172                                    .source(json));
173                    if (LOG.isDebugEnabled()) {
174                        try {
175                            if(LOG.isDebugEnabled()) {
176                                LOG.debug("Indexed " + entry.getKey().getClazz() + "(index:" + scm.getIndexName() + ",type:" + scm.getElasticTypeName() +
177                                    ") of id " + entry.getKey().getId() + " and source " + json.string());//new String(json, "UTF-8"));
178                            }
179                        } catch (IOException e) {}
180                    }
181                } finally {
182                    persistenceInterceptor.destroy();
183                }
184            }
185    
186            // Execute delete requests
187            for(IndexEntityKey key : toDelete) {
188                SearchableClassMapping scm = elasticSearchContextHolder.getMappingContextByType(key.getClazz());
189                bulkRequestBuilder.add(
190                            new DeleteRequest(scm.getIndexName())
191                                    .type(scm.getElasticTypeName())
192                                    .id(key.getId())
193                );
194            }
195    
196            if (bulkRequestBuilder.numberOfActions() > 0) {
197                try {
198                    bulkRequestBuilder.setRefresh(false).execute()
199                            .addListener(new OperationBatch(0, toIndex, toDelete));
200                } catch (Exception e) {
201                    throw new IndexException("Failed to index/delete " + bulkRequestBuilder.numberOfActions(), e);
202                }
203            }
204    
205        }
206    
207        class OperationBatch implements ActionListener<BulkResponse> {
208    
209            private int attempts;
210            private Map<IndexEntityKey, Object> toIndex;
211            private Set<IndexEntityKey> toDelete;
212    
213            OperationBatch(int attempts, Map<IndexEntityKey, Object> toIndex, Set<IndexEntityKey> toDelete) {
214                this.attempts = attempts;
215                this.toIndex = toIndex;
216                this.toDelete = toDelete;
217            }
218    
219            public void onResponse(BulkResponse bulkResponse) {
220                for(BulkItemResponse item : bulkResponse.items()) {
221                    boolean removeFromQueue = !item.isFailed()
222                            || item.getFailureMessage().indexOf("UnavailableShardsException") >= 0;
223                    // On shard failure, do not re-push.
224                    if (removeFromQueue) {
225                        // remove successful OR fatal ones.
226                        Class<?> entityClass = elasticSearchContextHolder.findMappedClassByElasticType(item.getType());
227                        if (entityClass == null) {
228                            LOG.error("Elastic type [" + item.getType() + "] is not mapped.");
229                            continue;
230                        }
231                        IndexEntityKey key = new IndexEntityKey(item.getId(), entityClass);
232                        toIndex.remove(key);
233                        toDelete.remove(key);
234                    }
235                    if (item.isFailed()) {
236                        LOG.error("Failed bulk item: " + item.getFailureMessage());
237                    }
238                }
239                if (!toIndex.isEmpty() || !toDelete.isEmpty()) {
240                    push();
241                } else {
242                    if(LOG.isDebugEnabled()) {
243                        LOG.debug("Batch complete: " + bulkResponse.items().length + " actions.");
244                    }
245                }
246            }
247    
248            public void onFailure(Throwable e) {
249                // Everything failed. Re-push all.
250                LOG.error("Bulk request failure", e);
251                push();
252            }
253    
254    
255            /**
256             * Push specified entities to be retried.
257             */
258            public void push() {
259                LOG.debug("Pushing retry: " + toIndex.size() + " indexing, " + toDelete.size() + " deletes.");
260                for (Map.Entry<IndexEntityKey, Object> entry : toIndex.entrySet()) {
261                    synchronized (this) {
262                        if (!indexRequests.containsKey(entry.getKey())) {
263                            // Do not overwrite existing stuff in the queue.
264                            indexRequests.put(entry.getKey(), entry.getValue());
265                        }
266                    }
267                }
268                for(IndexEntityKey key : toDelete) {
269                    synchronized (this) {
270                        if (!deleteRequests.contains(key)) {
271                            deleteRequests.add(key);
272                        }
273                    }
274                }
275    
276                executeRequests();
277            }
278        }
279    
280        class IndexEntityKey implements Serializable {
281    
282            /** stringified id. */
283            private final String id;
284            private final Class clazz;
285    
286            IndexEntityKey(String id, Class clazz) {
287                this.id = id;
288                this.clazz = clazz;
289            }
290    
291            IndexEntityKey(Object instance) {
292                this.clazz = GrailsHibernateUtil.unwrapIfProxy(instance).getClass();
293                SearchableClassMapping scm = elasticSearchContextHolder.getMappingContextByType(this.clazz);
294                if (scm == null) {
295                    throw new IllegalArgumentException("Class " + clazz + " is not a searchable domain class.");
296                }
297                this.id = (InvokerHelper.invokeMethod(instance, "ident", null)).toString();
298            }
299    
300            public String getId() {
301                return id;
302            }
303    
304            public Class getClazz() {
305                return clazz;
306            }
307    
308            @Override
309            public boolean equals(Object o) {
310                if (this == o) return true;
311                if (o == null || getClass() != o.getClass()) return false;
312    
313                IndexEntityKey that = (IndexEntityKey) o;
314    
315                if (!clazz.equals(that.clazz)) return false;
316                if (!id.equals(that.id)) return false;
317    
318                return true;
319            }
320    
321            @Override
322            public int hashCode() {
323                int result = id.hashCode();
324                result = 31 * result + clazz.hashCode();
325                return result;
326            }
327    
328            @Override
329            public String toString() {
330                return "IndexEntityKey{" +
331                        "id=" + id +
332                        ", clazz=" + clazz +
333                        '}';
334            }
335        }
