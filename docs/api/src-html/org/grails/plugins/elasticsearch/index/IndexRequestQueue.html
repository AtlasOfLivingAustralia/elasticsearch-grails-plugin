046    /**
047     * Holds objects to be indexed.
048     * <br/>
049     * It looks like we need to keep object references in memory until they indexed properly.
050     * If indexing fails, all failed objects are retried. Still no support for max number of retries (todo)
051     * NOTE: if cluster state is RED, everything will probably fail and keep retrying forever.
052     * NOTE: This is shared class, so need to be thread-safe.
053     */
054    public class IndexRequestQueue implements InitializingBean {
055    
056        private static final Logger LOG = Logger.getLogger(IndexRequestQueue.class);
057    
058        private JSONDomainFactory jsonDomainFactory;
059        private ElasticSearchContextHolder elasticSearchContextHolder;
060        private Client elasticSearchClient;
061        private HibernatePersistenceContextInterceptor persistenceInterceptor;
062        private SessionFactory sessionFactory;

071         */

111        public void addIndexRequest(Object instance, Serializable id) {
112            synchronized (this) {
113                IndexEntityKey key = id == null ? new IndexEntityKey(instance) :
114                        new IndexEntityKey(id.toString(), GrailsHibernateUtil.unwrapIfProxy(instance).getClass());
115                indexRequests.put(key, GrailsHibernateUtil.unwrapIfProxy(instance));
116            }
117        }
118    
119        public void addDeleteRequest(Object instance) {
120            synchronized (this) {
121                deleteRequests.add(new IndexEntityKey(instance));
122            }
123        }

125        public XContentBuilder toJSON(Object instance) {
126            try {
127                return jsonDomainFactory.buildJSON(instance);
128            } catch (Exception e) {
129                throw new IndexException("Failed to marshall domain instance [" + instance + "]", e);
130            }
131        }

133        /**
134         * Execute pending requests and clear both index & delete pending queues.
135         *
136         * @return Returns an OperationBatch instance which is a listener to the last executed bulk operation. Returns NULL
137         *         if there were no operations done on the method call.
138         */
139        public OperationBatch executeRequests() {
140            Map<IndexEntityKey, Object> toIndex = new LinkedHashMap<IndexEntityKey, Object>();
141            Set<IndexEntityKey> toDelete = new HashSet<IndexEntityKey>();
142    
143            cleanOperationBatchList();
144    
145            // Copy existing queue to ensure we are interfering with incoming requests.
146            synchronized (this) {
147                toIndex.putAll(indexRequests);
148                toDelete.addAll(deleteRequests);
149                indexRequests.clear();
150                deleteRequests.clear();
151            }
152    
153            // If there are domain instances that are both in the index requests & delete requests list,
154            // they are directly deleted.
155            toIndex.keySet().removeAll(toDelete);
156    
157            // If there is nothing in the queues, just stop here
158            if (toIndex.isEmpty() && toDelete.isEmpty()) {
159                return null;
160            }
161    
162            BulkRequestBuilder bulkRequestBuilder = elasticSearchClient.prepareBulk();
163            //bulkRequestBuilder.setRefresh(true);
164    
165            // Execute index requests
166            for (Map.Entry<IndexEntityKey, Object> entry : toIndex.entrySet()) {
167                SearchableClassMapping scm = elasticSearchContextHolder.getMappingContextByType(entry.getKey().getClazz());
168                persistenceInterceptor.init();
169                try {
170                    Session session = SessionFactoryUtils.getSession(sessionFactory, true);
171                    Object entity = entry.getValue();
172    
173                    // If this not a transient instance, reattach it to the session
174                    if (session.contains(entity)) {
175                        session.lock(entity, LockMode.NONE);
176                        LOG.debug("Reattached entity to session");
177                    }
178    
179                    XContentBuilder json = toJSON(entity);
180    
181                    bulkRequestBuilder.add(
182                            elasticSearchClient.prepareIndex()
183                                    .setIndex(scm.getIndexName())
184                                    .setType(scm.getElasticTypeName())
185                                    .setId(entry.getKey().getId()) // TODO : Composite key ?
186                                    .setSource(json)
187                    );
188                    if (LOG.isDebugEnabled()) {
189                        try {
190                            LOG.debug("Indexing " + entry.getKey().getClazz() + "(index:" + scm.getIndexName() + ",type:" + scm.getElasticTypeName() +
191                                    ") of id " + entry.getKey().getId() + " and source " + json.string());
192                        } catch (IOException e) {
193                        }
194                    }
195                } finally {
196                    persistenceInterceptor.destroy();
197                }
198            }
199    
200            // Execute delete requests
201            for (IndexEntityKey key : toDelete) {
202                SearchableClassMapping scm = elasticSearchContextHolder.getMappingContextByType(key.getClazz());
203                if (LOG.isDebugEnabled()) {
204                    LOG.debug("Deleting object from index " + scm.getIndexName() + " and type " + scm.getElasticTypeName() + " and ID " + key.getId());
205                }
206                bulkRequestBuilder.add(
207                        elasticSearchClient.prepareDelete()
208                                .setIndex(scm.getIndexName())
209                                .setType(scm.getElasticTypeName())
210                                .setId(key.getId())
211                );
212            }
213    
214            // Perform bulk request
215            OperationBatch completeListener = null;
216            if (bulkRequestBuilder.numberOfActions() > 0) {
217                completeListener = new OperationBatch(0, toIndex, toDelete);
218                operationBatchList.add(completeListener);
219                try {
220                    bulkRequestBuilder.execute().addListener(completeListener);
221                } catch (Exception e) {
222                    throw new IndexException("Failed to index/delete " + bulkRequestBuilder.numberOfActions(), e);
223                }
224            }
225    
226            return completeListener;
227        }
228    
229        public void waitComplete() {
230            LOG.debug("IndexRequestQueue.waitComplete() called");
231            List<OperationBatch> clone = new LinkedList<OperationBatch>();
232            synchronized (this) {
233                clone.addAll(operationBatchList);
234                operationBatchList.clear();
235            }
236            for (OperationBatch op : clone) {
237                op.waitComplete();
238            }
239        }
240    
241        private void cleanOperationBatchList() {
242            for (Iterator<OperationBatch> it = operationBatchList.iterator(); it.hasNext(); ) {
243                OperationBatch current = it.next();
244                if (current.isComplete()) {
245                    it.remove();
246                }
247            }
248            LOG.debug("OperationBatchList cleaned");
249        }

251        class OperationBatch implements ActionListener<BulkResponse> {
252    
253            private int attempts;
254            private Map<IndexEntityKey, Object> toIndex;
255            private Set<IndexEntityKey> toDelete;
256            private boolean complete = false;
257    
258            OperationBatch(int attempts, Map<IndexEntityKey, Object> toIndex, Set<IndexEntityKey> toDelete) {
259                this.attempts = attempts;
260                this.toIndex = toIndex;
261                this.toDelete = toDelete;
262            }
263    
264            public boolean isComplete() {
265                return complete;
266            }
267    
268            public void waitComplete() {
269                waitComplete(null);
270            }
271    
272            /**
273             * Wait for the operation to complete. Use this method to synchronize the application with the last ES operation.
274             *
275             * @param msTimeout A maximum timeout (in milliseconds) before the wait method returns, whether the operation has been completed or not.
276             *                  Default value is 5000 milliseconds
277             */
278            public void waitComplete(Integer msTimeout) {
279                msTimeout = msTimeout == null ? 5000 : msTimeout;
280                Long startTime = new Date().getTime();
281                Long currentTime = startTime;
282                try {
283                    while (!complete && currentTime - startTime < msTimeout) {
284                        Thread.sleep(100);
285                        currentTime = new Date().getTime();
286                    }
287                } catch (InterruptedException ie) {
288                }
289            }
290    
291            public void fireComplete() {
292                complete = true;
293            }
294    
295            public void onResponse(BulkResponse bulkResponse) {
296                for (BulkItemResponse item : bulkResponse.items()) {
297                    boolean removeFromQueue = !item.isFailed()
298                            || item.getFailureMessage().indexOf("UnavailableShardsException") >= 0;
299                    // On shard failure, do not re-push.
300                    if (removeFromQueue) {
301                        // remove successful OR fatal ones.
302                        Class<?> entityClass = elasticSearchContextHolder.findMappedClassByElasticType(item.getType());
303                        if (entityClass == null) {
304                            LOG.error("Elastic type [" + item.getType() + "] is not mapped.");
305                            continue;
306                        }
307                        IndexEntityKey key = new IndexEntityKey(item.getId(), entityClass);
308                        toIndex.remove(key);
309                        toDelete.remove(key);
310                    }
311                    if (item.isFailed()) {
312                        LOG.error("Failed bulk item: " + item.getFailureMessage());
313                    }
314                }
315                if (!toIndex.isEmpty() || !toDelete.isEmpty()) {
316                    push();
317                } else {
318                    fireComplete();
319                    if (LOG.isDebugEnabled()) {
320                        LOG.debug("Batch complete: " + bulkResponse.items().length + " actions.");
321                    }
322                }
323            }
324    
325            public void onFailure(Throwable e) {
326                // Everything failed. Re-push all.
327                LOG.error("Bulk request failure", e);
328                push();
329            }
330    
331    
332            /**
333             * Push specified entities to be retried.
334             */
335            public void push() {
336                LOG.debug("Pushing retry: " + toIndex.size() + " indexing, " + toDelete.size() + " deletes.");
337                for (Map.Entry<IndexEntityKey, Object> entry : toIndex.entrySet()) {
338                    synchronized (this) {
339                        if (!indexRequests.containsKey(entry.getKey())) {
340                            // Do not overwrite existing stuff in the queue.
341                            indexRequests.put(entry.getKey(), entry.getValue());
342                        }
343                    }
344                }
345                for (IndexEntityKey key : toDelete) {
346                    synchronized (this) {
347                        if (!deleteRequests.contains(key)) {
348                            deleteRequests.add(key);
349                        }
350                    }
351                }
352    
353                executeRequests();
354            }
355        }

357        class IndexEntityKey implements Serializable {
358    
359            /**
360             * stringified id.
361             */
362            private final String id;
363            private final Class clazz;
364    
365            IndexEntityKey(String id, Class clazz) {
366                this.id = id;
367                this.clazz = clazz;
368            }
369    
370            IndexEntityKey(Object instance) {
371                this.clazz = GrailsHibernateUtil.unwrapIfProxy(instance).getClass();
372                SearchableClassMapping scm = elasticSearchContextHolder.getMappingContextByType(this.clazz);
373                if (scm == null) {
374                    throw new IllegalArgumentException("Class " + clazz + " is not a searchable domain class.");
375                }
376                this.id = (InvokerHelper.invokeMethod(instance, "ident", null)).toString();
377            }
378    
379            public String getId() {
380                return id;
381            }
382    
383            public Class getClazz() {
384                return clazz;
385            }
386    
387            @Override
388            public boolean equals(Object o) {
389                if (this == o) return true;
390                if (o == null || getClass() != o.getClass()) return false;
391    
392                IndexEntityKey that = (IndexEntityKey) o;
393    
394                if (!clazz.equals(that.clazz)) return false;
395                if (!id.equals(that.id)) return false;
396    
397                return true;
398            }
399    
400            @Override
401            public int hashCode() {
402                int result = id.hashCode();
403                result = 31 * result + clazz.hashCode();
404                return result;
405            }
406    
407            @Override
408            public String toString() {
409                return "IndexEntityKey{" +
410                        "id=" + id +
411                        ", clazz=" + clazz +
412                        '}';
413            }
414        }
