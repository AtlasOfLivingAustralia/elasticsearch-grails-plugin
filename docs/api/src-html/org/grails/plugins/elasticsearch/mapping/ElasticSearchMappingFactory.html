016    package org.grails.plugins.elasticsearch.mapping;
017    
018    import org.codehaus.groovy.grails.commons.GrailsClassUtils;
019    import org.grails.plugins.elasticsearch.ElasticSearchContextHolder;
020    import org.springframework.util.ClassUtils;
021    
022    import java.util.*;
023    
024    /**
025     * Build ElasticSearch class mapping based on attributes provided by closure.
026     */
027    public class ElasticSearchMappingFactory {
028    
029        private static final Set<String> SUPPORTED_FORMAT = new HashSet<String>(Arrays.asList(
030                "string", "integer", "long", "float", "double", "boolean", "null", "date"));
031    
032        private static Class JODA_TIME_BASE;
033    
034        static {
035            try {
036                JODA_TIME_BASE = Class.forName("org.joda.time.ReadableInstant");
037            } catch (ClassNotFoundException e) { }
038        }
039    
040        public static Map<String, Object> getElasticMapping(SearchableClassMapping scm) {
041            Map<String, Object> elasticTypeMappingProperties = new LinkedHashMap<String, Object>();
042    
043            if (!scm.isAll()) {
044                // "_all" : {"enabled" : true}
045                elasticTypeMappingProperties.put("_all",
046                    Collections.singletonMap("enabled", false));
047            }
048    
049            // Map each domain properties in supported format, or object for complex type
050            for(SearchableClassPropertyMapping scpm : scm.getPropertiesMapping()) {
051                // Does it have custom mapping?
052                String propType = scpm.getGrailsProperty().getTypePropertyName();
053                Map<String, Object> propOptions = new LinkedHashMap<String, Object>();
054                // Add the custom mapping (searchable static property in domain model)
055                propOptions.putAll(scpm.getAttributes());
056                if (!(SUPPORTED_FORMAT.contains(scpm.getGrailsProperty().getTypePropertyName()))) {
057                    // Handle embedded persistent collections, ie List<String> listOfThings
058                    if (scpm.getGrailsProperty().isBasicCollectionType()) {
059                        String basicType = ClassUtils.getShortName(scpm.getGrailsProperty().getReferencedPropertyType()).toLowerCase(Locale.ENGLISH);
060                        if (SUPPORTED_FORMAT.contains(basicType)) {
061                            propType = basicType;
062                        }
063                    // Handle arrays
064                    } else if (scpm.getGrailsProperty().getReferencedPropertyType().isArray()) {
065                        String basicType = ClassUtils.getShortName(scpm.getGrailsProperty().getReferencedPropertyType().getComponentType()).toLowerCase(Locale.ENGLISH);
066                        if (SUPPORTED_FORMAT.contains(basicType)) {
067                            propType = basicType;
068                        }
069                    } else if (isDateType(scpm.getGrailsProperty().getReferencedPropertyType())) {
070                        propType = "date";
071                    } else if (GrailsClassUtils.isJdk5Enum(scpm.getGrailsProperty().getReferencedPropertyType())) {
072                        propType = "string";
073                    } else if (scpm.getConverter() != null) {
074                        // Use 'string' type for properties with custom converter.
075                        // Arrays are automatically resolved by ElasticSearch, so no worries.
076                        propType = "string";
077                    } else {
078                        propType = "object";
079                    }
080    
081                    if (scpm.getReference() != null) {
082                        propType = "object";      // fixme: think about composite ids.
083                    } else if (scpm.isComponent()) {
084                        // Proceed with nested mapping.
085                        // todo limit depth to avoid endless recursion?
086                        propType = "object";
087                        //noinspection unchecked
088                        propOptions.putAll((Map<String, Object>)
089                                (getElasticMapping(scpm.getComponentPropertyMapping()).values().iterator().next()));
090    
091                    }
092    
093                    // Once it is an object, we need to add id & class mappings, otherwise
094                    // ES will fail with NullPointer.
095                    if (scpm.isComponent() || scpm.getReference() != null) {
096                        @SuppressWarnings({"unchecked"})
097                        Map<String, Object> props = (Map<String, Object>) propOptions.get("properties");
098                        if (props == null) {
099                            props = new LinkedHashMap<String, Object>();
100                            propOptions.put("properties", props);
101                        }
102                        props.put("id", defaultDescriptor("long", "no", true));
103                        props.put("class", defaultDescriptor("string", "no", true));
104                        props.put("ref", defaultDescriptor("string", "no", true));
105                    }
106                }
107                propOptions.put("type", propType);
108                // See http://www.elasticsearch.com/docs/elasticsearch/mapping/all_field/
109                if (!propType.equals("object") && scm.isAll()) {
110                    // does it make sense to include objects into _all?
111                    if (scpm.shouldExcludeFromAll()) {
112                        propOptions.put("include_in_all", false);
113                    } else {
114                        propOptions.put("include_in_all", true);
115                    }
116                }
117                // todo only enable this through configuration...
118                if (propType.equals("string") && scpm.isAnalyzed()) {
119                    propOptions.put("term_vector", "with_positions_offsets");
120                }
121                elasticTypeMappingProperties.put(scpm.getPropertyName(), propOptions);
122            }
123    
124            Map<String, Object> mapping = new LinkedHashMap<String, Object>();
125            mapping.put(scm.getElasticTypeName(),
126                    Collections.singletonMap("properties", elasticTypeMappingProperties));
127    
128            return mapping;
129        }
130    
131        private static boolean isDateType(Class type) {
132            return (JODA_TIME_BASE != null && JODA_TIME_BASE.isAssignableFrom(type)) || java.util.Date.class.isAssignableFrom(type);
133        }
134    
135        private static Map<String, Object> defaultDescriptor(String type, String index, boolean excludeFromAll) {
136            Map<String, Object> props = new LinkedHashMap<String, Object>();
137            props.put("type", type);
138            props.put("index", index);
139            props.put("include_in_all", !excludeFromAll);
140            return props;
141        }
142    
143    }
