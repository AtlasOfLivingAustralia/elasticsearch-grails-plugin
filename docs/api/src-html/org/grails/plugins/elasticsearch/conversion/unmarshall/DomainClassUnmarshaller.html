052        public Collection buildResults(SearchHits hits) {
053            DefaultUnmarshallingContext unmarshallingContext = new DefaultUnmarshallingContext();
054            List results = new ArrayList();
055            for(SearchHit hit : hits) {
056                String domainClassName = hit.index().equals(hit.type()) ? DefaultGroovyMethods.capitalize(hit.index()) : (hit.index() + '.' + DefaultGroovyMethods.capitalize(hit.type()));
057                SearchableClassMapping scm = elasticSearchContextHolder.getMappingContext(domainClassName);
058                if (scm == null) {
059                    LOG.warn("Unknown SearchHit: " + hit.id() + "#" + hit.type() + ", domain class name: " + domainClassName);
060                    continue;
061                }
062    
063                GrailsDomainClassProperty identifier = scm.getDomainClass().getIdentifier();
064                Object id = typeConverter.convertIfNecessary(hit.id(), identifier.getType());
065                GroovyObject instance = (GroovyObject) scm.getDomainClass().newInstance();
066                instance.setProperty(identifier.getName(), id);
067    
068                /*def mapContext = elasticSearchContextHolder.getMappingContext(domainClass.propertyName)?.propertiesMapping*/
069                Map rebuiltProperties = new HashMap();
070                for(Map.Entry<String, Object> entry : hit.getSource().entrySet()) {
071                    unmarshallingContext.getUnmarshallingStack().push(entry.getKey());
072                    rebuiltProperties.put(entry.getKey(),
073                            unmarshallProperty(scm.getDomainClass(), entry.getKey(), entry.getValue(), unmarshallingContext));
074                    populateCyclicReference(instance, rebuiltProperties, unmarshallingContext);
075                    unmarshallingContext.resetContext();
076                }
077                // todo manage read-only transient properties...
078                bind.invoke(instance, "bind", new Object[] { instance, rebuiltProperties });
079    
080                results.add(instance);
081            }
082            return results;
083        }
084    
085        private void populateCyclicReference(Object instance, Map<String, Object> rebuiltProperties, DefaultUnmarshallingContext unmarshallingContext) {
086            for(CycleReferenceSource cr : unmarshallingContext.getCycleRefStack()) {
087                populateProperty(cr.getCyclePath(), rebuiltProperties, resolvePath(cr.getSourcePath(), instance, rebuiltProperties));
088            }
089        }
090    
091        private Object resolvePath(String path, Object instance, Map<String, Object> rebuiltProperties) {
092            if (path == null || path.equals("")) {
093                return instance;
094            } else {
095                StringTokenizer st = new StringTokenizer(path, "/");
096                Object currentProperty = rebuiltProperties;
097                while (st.hasMoreTokens()) {
098                    String part = st.nextToken();
099                    try {
100                        int index = Integer.parseInt(part);
101                        currentProperty = DefaultGroovyMethods.getAt(DefaultGroovyMethods.asList((Collection) currentProperty), index);
102                    } catch (NumberFormatException e) {
103                        currentProperty = DefaultGroovyMethods.getAt(currentProperty, part);
104                    }
105                }
106                return currentProperty;
107            }
108        }
109    
110        private void populateProperty(String path, Map<String, Object> rebuiltProperties, Object value) {
111            String last = null;
112            Object currentProperty = rebuiltProperties;
113            StringTokenizer st = new StringTokenizer(path, "/");
114            int size = st.countTokens();
115            int index = 0;
116            while (st.hasMoreTokens()) {
117                String part = st.nextToken();
118                if (index < size - 1) {
119                    try {
120                        if (currentProperty instanceof Collection) {
121                            //noinspection unchecked
122                            currentProperty = DefaultGroovyMethods.getAt(((Collection<Object>) currentProperty).iterator(), DefaultGroovyMethods.toInteger(part));
123                        } else {
124                            currentProperty = DefaultGroovyMethods.getAt(currentProperty, part);
125                        }
126                    } catch (Exception e) {
127                        LOG.warn("/!\\ Error when trying to populate " + path);
128                        LOG.warn("Cannot get " + part + " on " + currentProperty + " from " + rebuiltProperties);
129                        e.printStackTrace();
130                    }
131                }
132                if (!st.hasMoreTokens()) {
133                    last = part;
134                }
135                index++;
136            }
137            try {
138                Integer.parseInt(last);
139                ((Collection) currentProperty).add(value);
140            } catch (NumberFormatException e) {
141                DefaultGroovyMethods.putAt(currentProperty, last, value);
142            }
143        }

145        private Object unmarshallProperty(GrailsDomainClass domainClass, String propertyName, Object propertyValue, DefaultUnmarshallingContext unmarshallingContext) {
146            // TODO : adapt behavior if the mapping option "component" or "reference" are set
147            // below is considering the "component" behavior
148            SearchableClassPropertyMapping scpm = elasticSearchContextHolder.getMappingContext(domainClass).getPropertyMapping(propertyName);
149            Object parseResult = null;
150            if (propertyValue instanceof Map) {
151    
152                @SuppressWarnings({"unchecked"})
153                Map<String, Object> data = (Map<String, Object>) propertyValue;
154    
155                // Handle cycle reference
156                if (data.containsKey("ref")) {
157                    unmarshallingContext.addCycleRef(propertyValue);
158                    return null;
159                }
160    
161                // Searchable reference.
162                if (scpm.getReference() != null) {
163                    Class<?> refClass = scpm.getBestGuessReferenceType();
164                    GrailsDomainClass refDomainClass = null;
165                    for(GrailsClass dClazz : grailsApplication.getArtefacts(DomainClassArtefactHandler.TYPE)) {
166                        if (dClazz.getClazz().equals(refClass)) {
167                            refDomainClass = (GrailsDomainClass) dClazz;
168                            break;
169                        }
170                    }
171                    if (refDomainClass == null) {
172                        throw new IllegalStateException("Found reference to non-domain class: " + refClass);
173                    }
174                    return unmarshallReference(refDomainClass, data, unmarshallingContext);
175                }
176    
177                if (data.containsKey("class")) {
178                    // Embedded instance.
179                    if (!scpm.isComponent()) {
180                        // maybe ignore?
181                        throw new IllegalStateException("Property " + domainClass.getName() + "." + propertyName +
182                                    " is not mapped as [component], but broken search hit found.");
183                    }
184                    GrailsDomainClass nestedDomainClass = (GrailsDomainClass)
185                            grailsApplication.getArtefact(DomainClassArtefactHandler.TYPE, (String) data.get("class"));
186                    if (domainClass != null) {
187                        // Unmarshall 'component' instance.
188                        if (!scpm.isComponent()) {
189                            throw new IllegalStateException("Object " + data.get("class") +
190                                    " found in index, but [" + propertyName + "] is not mapped as component.");
191                        }
192                        parseResult = unmarshallDomain(nestedDomainClass, data.get("id"), data, unmarshallingContext);
193                    }
194                }
195            } else if (propertyValue instanceof Collection) {
196                List<Object> results = new ArrayList<Object>();
197                int index = 0;
198                for(Object innerValue : (Collection) propertyValue) {
199                    unmarshallingContext.getUnmarshallingStack().push(String.valueOf(index));
200                    Object parseItem = unmarshallProperty(domainClass, propertyName, innerValue, unmarshallingContext);
201                    if (parseItem != null) {
202                        results.add(parseItem);
203                    }
204                    index++;
205                    unmarshallingContext.getUnmarshallingStack().pop();
206                }
207                parseResult = results;
208            } else {
209                // consider any custom property editors here.
210                if (scpm.getConverter() != null) {
211                    if (scpm.getConverter() instanceof Class) {
212                        try {
213                            PropertyEditor propertyEditor = (PropertyEditor) ((Class) scpm.getConverter()).newInstance();
214                            propertyEditor.setAsText((String) propertyValue);
215                            parseResult = propertyEditor.getValue();
216                        } catch (Exception e) {
217                            throw new IllegalArgumentException("Unable to unmarshall " + propertyName +
218                                    " using " + scpm.getConverter(), e);
219                        }
220                    }
221                } else if (scpm.getReference() != null) {
222    
223                    // This is a reference and it MUST be null because it's not a Map.
224                    if (propertyValue != null) {
225                        throw new IllegalStateException("Found searchable reference which is not a Map: " + domainClass + "." + propertyName +
226                                " = " + propertyValue);
227                    }
228    
229                    parseResult = null;
230                }
231            }
232            if (parseResult != null) {
233                return parseResult;
234            } else {
235                return propertyValue;
236            }
237        }

240        private Object unmarshallReference(GrailsDomainClass domainClass, Map<String, Object> data, DefaultUnmarshallingContext unmarshallingContext) {
241            // As a simplest scenario recover object directly from ElasticSearch.
242            // todo add first-level caching and cycle ref checking
243            String indexName = elasticSearchContextHolder.getMappingContext(domainClass).getIndexName();
244            String name = elasticSearchContextHolder.getMappingContext(domainClass).getElasticTypeName();
245            // A property value is expected to be a map in the form [id:ident]
246            Object id = data.get("id");
247            GetResponse response = elasticSearchClient.get(new GetRequest(indexName)
248                    .operationThreaded(false)
249                    .type(name)
250                    .id(typeConverter.convertIfNecessary(id, String.class)))
251                    .actionGet();
252            return unmarshallDomain(domainClass, response.id(), response.sourceAsMap(), unmarshallingContext);
253        }
254    
255    
256        private Object unmarshallDomain(GrailsDomainClass domainClass, Object providedId, Map<String, Object> data, DefaultUnmarshallingContext unmarshallingContext) {
257            GrailsDomainClassProperty identifier = domainClass.getIdentifier();
258            Object id = typeConverter.convertIfNecessary(providedId, identifier.getType());
259            GroovyObject instance = (GroovyObject) domainClass.newInstance();
260            instance.setProperty(identifier.getName(), id);
261            for(Map.Entry<String, Object> entry : data.entrySet()) {
262                if (!entry.getKey().equals("class") && !entry.getKey().equals("id")) {
263                    unmarshallingContext.getUnmarshallingStack().push(entry.getKey());
264                    Object propertyValue = unmarshallProperty(domainClass, entry.getKey(), entry.getValue(), unmarshallingContext);
265                    bind.invoke(instance, "bind", new Object[] { instance, Collections.singletonMap(entry.getKey(), propertyValue) });
266                    unmarshallingContext.getUnmarshallingStack().pop();
267                }
268            }
269            return instance;
270        }
